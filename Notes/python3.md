# Make sure you have latest version of Python (at least 3.11.x)
# To check python version
```
python3 --version
```

# To check where python executable is available
```
which python3
```

# To create a venv
```
python3 -m venv .venv
source .venv/bin/activate
```

# To come out of a venv
```
deactivate
```
